\def\SIR{
    \begin{table}[H]
        \centering
        \caption{SIR模型参数表}
        \begin{tabular}{ll}
            \hline
            符号     & 含义         \\
            \hline
            $\alpha$ & \PText{S}{I} \\
            $\beta$  & \PText{I}{R} \\
            \hline
        \end{tabular}
    \end{table}
    \def\SI{IS\alpha}
    \def\IR{I\beta}
    \begin{align}
        \dt{S} & = -\SI      \\
        \dt{I} & = \SI - \IR \\
        \dt{R} & = \IR
    \end{align}
}
\def\SEIR{
    \begin{table}[H]

            \hline
            符号       & 含义         \\
            \hline
            $\alpha$   & \PText{S}{E} \\
            $\gamma$   & \PText{E}{I} \\
            $\epsilon$ & \PText{E}{R} \\
            $\beta$    & \PText{I}{R} \\
            \hline
        \end{tabular}
    \end{table}
    \def\SE{(I+E)S\alpha}
    \def\EI{E\gamma}
    \def\ER{E\epsilon}
    \def\IR{I\beta}
    \begin{align}
        \dt{S} & = -\SE            \\
        \dt{E} & = \SE - \EI - \ER \\
        \dt{I} & = \EI - \IR       \\
        \dt{R} & = \IR +\ER
    \end{align}
}
